\begin{block}{The off-shell Higgs boson}
The off-shell region of the Higgs boson is enhanced in VV decays above $2m_\text{V}$,
and is characterized by a strong interference component.
The hallmark of the off-shell region is the independence from the the width,
% as $\sim 4$ MeV $<<$ $2m_\text{V}$.
{\begin{align}
  \frac{\sigma(\text{i} \to \text{H} \to \text{f})}{ds} \propto \frac{1}{(s-M_\text{H}^2)^2 + M_\text{H}^2\Gamma_\text{H}^2}\\
  \left(\sqrt{s} - M_\text{H}\right) >> \Gamma_H \text{ when $\sqrt{s} \geq 2m_\text{V}$} \nonumber
\end{align}}
\end{block}

\begin{block}{Higgs Substructure}
We probe a nucleon form factor through the threshold energy $\Lambda_\text{H}$
that tests Higgs boson substructure.
% Figure~\ref{fig:structure} showcases the evolution given different values of $\Lambda_\text{H}$
% for the form factor described in Eq.~\ref{eq:formFactor}, where $\Lambda_\text{H}$ is the
% energy scale of the form factor and $d$ the characteristic size, 
% which can be with the scan shown in Fig.~\ref{fig:structureScan}.
{\begin{align*}
  F(q^2) &= 
     \left[
     \frac{1}{1 + | q^2 | / \Lambda_\text{H}^2}
     \right]^2
     \label{eq:formFactor}, &
    d &= \frac{\hbar c}{\Lambda_\text{H}}
\end{align*}}

\begin{block}{Yukawa Couplings}
Scans over $\kappa_\text{t,Q}, \widetilde{\kappa}_\text{t,Q}$ in the EFT limit for gluon-fusion production,
where $\kappa_\text{Q} \propto C_\text{HG}$ and $\tilde{\kappa}_\text{Q} \propto \tilde{C}_\text{HG}$.
CP-odd sensitivity comes from interference (or lack thereof) with SM background in off-shell.
\begin{figure}[!htb]
  \centering
  \includegraphics[width=\linewidth]{images/Kappa_QTilde.pdf}
\end{figure}

Scans over u,d,s, and c quarks build off of previous work and utilize off-shell to remove
constraints on vector boson couplings.
\begin{figure}[!htb]
  \centering
  \includegraphics[width=1.01\linewidth]{images/Kappa_c.pdf}
\end{figure}

\end{block}
